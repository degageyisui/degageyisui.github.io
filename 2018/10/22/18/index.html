SCOI2011 糖果

Description

幼儿园里有N个小朋友，lxhgww老师现在想要给这些小朋友们分配糖果，要求每个小朋友都要分到糖果。但是小朋友们也有嫉妒心，总是会提出一些要求，比如小明不希望小红分到的糖果比他的多，于是在分配糖果的时候，lxhgww需要满足小朋友们的K个要求。幼儿园的糖果总是有限的，lxhgww想知道他至少需要准备多少个糖果，才能使得每个小朋友都能够分到糖果，并且满足小朋友们所有的要求。

Input

输入的第一行是两个整数N，K。接下来K行，表示这些点需要满足的关系，每行3个数字，X，A，B。如果X=1， 表示第A个小朋友分到的糖果必须和第B个小朋友分到的糖果一样多；如果X=2， 表示第A个小朋友分到的糖果必须少于第B个小朋友分到的糖果；如果X=3， 表示第A个小朋友分到的糖果必须不少于第B个小朋友分到的糖果；如果X=4， 表示第A个小朋友分到的糖果必须多于第B个小朋友分到的糖果；如果X=5， 表示第A个小朋友分到的糖果必须不多于第B个小朋友分到的糖果；

Output

输出一行，表示lxhgww老师至少需要准备的糖果数，如果不能满足小朋友们的所有要求，就输出-1。

Sample Input

5 7
1 1 2
2 3 2
4 4 1
3 4 5
5 4 5
2 3 5
4 5 1

Sample Output

11

Hint

11
说明
【数据范围】

对于30%的数据，保证 N<=100

对于100%的数据，保证 N<=100000

对于所有的数据，保证 K<=100000，1<=X<=5，1<=A, B<=N

Solution

首先，这道题要看出是一个差分约束系统，而求至少便是求最小值，所以应转化为$d[j]>=d[i]+Wij$，这样跑最长路(只有用spfa)并且判0环就行了，然而发现这样只会有60，
因为数据大和毒，导致判0环会炸。然后发现可以强联通分量缩点，那么每个分量的值肯定是一样的，如果边中有1就说明不符合。最后加一个超级起点，从它开始拓扑加dp就能过了。

上代码：

#include<bits/stdc++.h>
#define maxn 150005
using namespace std;
int n,m,op,a,b;
typedef long long ll;
struct eage{
    int to,next,len;
}e[maxn<<1];
int np=0,first[maxn];
void add(int u,int v,int len){
    e[++np]=(eage){v,first[u],len};
    first[u]=np;
}
ll d[maxn];
bool vis[maxn];
int st[maxn],top=0,low[maxn],dfn[maxn],clock_=0,belong[maxn],cc=0;
bool ok=false;
int stt[maxn],topp=0;
bool run(){
    for(int i=1;i<=topp;i++){
        int t=stt[i];
        for(int p=first[t];p;p=e[p].next){
            int j=e[p].to,c=e[p].len;
            if(belong[j]==belong[t]&&c==1) return true;
        }
    }
    return false;
}
void dfs(int i){
    vis[i]=1;
    dfn[i]=low[i]=++clock_;
    st[++top]=i;
    for(int p=first[i];p;p=e[p].next){
        int j=e[p].to;
        if(vis[j]){
            if(!belong[j]) low[i]=min(low[i],dfn[j]);
            continue;
        }
        dfs(j);
        low[i]=min(low[i],low[j]);
    }
    if(low[i]==dfn[i]){
        cc++;topp=0;
        while(1){
            int t=st[top--];
            belong[t]=cc;
            stt[++topp]=t;
            if(t==i) break;
        } 
        bool okk=run(); 
        if(okk==true) ok=true;
    }
} 
struct node{
    int to;ll len;
};
vector<node>g[maxn];
queue<int>q; 
int rd[maxn];
void work(){
    for(int i=1;i<=n+1;i++){
        for(int p=first[i];p;p=e[p].next){
            int j=e[p].to;ll c=e[p].len;
            if(belong[j]!=belong[i]) g[belong[i]].push_back((node){belong[j],c}),rd[belong[j]]++;
        }
    }

    d[belong[n+1]]=0;
    q.push(belong[n+1]);
    while(!q.empty()){
        int i=q.front();q.pop();
        for(int j=0;j<g[i].size();j++){
            node t=g[i][j];
            int kk=t.to;ll c=t.len;
            d[kk]=max(d[kk],d[i]+c);
            rd[kk]--;
            if(rd[kk]==0) q.push(kk);
        }
    }
}
void init(){
    scanf("%d%d",&n,&m);
    d[n+1]=0;
    for(int i=1;i<=m;i++){
        scanf("%d%d%d",&op,&a,&b);
        if(op==1){
            add(a,b,0);
            add(b,a,0);
        }
        else if(op==2){
            add(a,b,1);
        }
        else if(op==3){
            add(b,a,0);
        }
        else if(op==4){
            add(b,a,1);
        }
        else if(op==5){
            add(a,b,0);
        }
    }    
    for(int i=1;i<=n;i++) add(n+1,i,1);
    for(int i=1;i<=n+1;i++){
        if(!vis[i]) dfs(i);
    }
    if(ok) printf("-1");
    else{
        work();
        ll ans=0;
        for(int i=1;i<=n+1;i++) ans+=d[belong[i]];
        printf("%lld",ans);
    }
}
int main(){
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout); 
    init();

    return 0;
}
