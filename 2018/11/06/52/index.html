Description

一个n行m列的矩阵，每个为0或1，如果一个边长为K的正方形包含的所有数都为1，那么会形成一个强度为K的电场，有q次询问，每次询问会把1个1变做0，求每次询问的最大电场强度值。

Hint

$n,m<=2 \ast 10^3 ,1<=q<=10^4$

Sample Input

5 5 5
+-+++
+++++
+++++
+++++
++++-
1 5
2 2
5 3
2 3
1 1

Sample Output

4
3
3
2
2

Solution

首先会想到用单调栈$n^2$计算最大1正方形，然后发现每改一个只会影响一列的up数组，然后发现如果答案递减不好维护，那么正难则反，我们先将所有的改动加进来，然后倒序考虑所有操作。
那么答案是会依次递增的，我们维护一个up和一个down数组来记录每个格子向上和向下最多多少格，对于每次询问，O(N)更新那一列的up和down，再O(n)单调队列扫一遍那一行，类似滑动窗口的来查询长度为k的序列的$min(up_i)和min(down_i)$，看它们加起来是否大于等于k，如果满足继续扩大k，否则标记答案。

上代码：

#include<bits/stdc++.h>
#define maxn 2005 
#define maxq 10005
#define inf 0x3f3f3f3f
using namespace std;
struct node{
    int x,y,ans;
}nd[maxq];
int n,m,a[maxn][maxn],q,st[maxn],top=0,up[maxn][maxn],down[maxn][maxn];
char s[maxn];
int maxx=0;
void calc(){
    for(int j=1;j<=m;j++)
    for(int i=1;i<=n;i++) up[i][j]=(a[i][j]==0?0:up[i-1][j]+1); 
    for(int j=1;j<=m;j++)
    for(int i=n;i>=1;i--) down[i][j]=(a[i][j]==0?0:down[i+1][j]+1);

    st[0]=0;
    for(int i=1;i<=n;i++){
        top=0;
        for(int j=1;j<=m;j++){
            while(top&&up[i][j]<=up[i][st[top]]){//单调栈 
                maxx=max(maxx,min(up[i][st[top]],j-1-st[top-1]));
                top--;
            }
            st[++top]=j;
        }
        while(top){
            maxx=max(maxx,min(up[i][st[top]],m-st[top-1]));
            top--;
        }
    }
}
void calcc(int j){
    for(int i=1;i<=n;i++) up[i][j]=(a[i][j]==0?0:up[i-1][j]+1); 
    for(int i=n;i>=1;i--) down[i][j]=(a[i][j]==0?0:down[i+1][j]+1);
}
int q1[maxn],q2[maxn],front1,front2,rear1,rear2;
bool check(int i,int len){
    if(len>min(n,m)) return false;
    front1=front2=rear1=rear2=0;
    for(int j=1;j<=m;j++){
        while(front1<rear1&&j-q1[front1]>=len)front1++;
        while(front1<rear1&&up[i][q1[rear1-1]]>=up[i][j])rear1--;
        q1[rear1++]=j;
        while(front2<rear2&&j-q2[front2]>=len)front2++;
        while(front2<rear2&&down[i][q2[rear2-1]]>=down[i][j])rear2--;
        q2[rear2++]=j;
        if(j>=len&&up[i][q1[front1]]+down[i][q2[front2]]-1>=len)return true;
    }
    return false;
}
void init(){
    scanf("%d%d%d",&n,&m,&q);
    for(int i=1;i<=n;i++){
        scanf("%s",s+1);
        for(int j=1;j<=m;j++) a[i][j]=(s[j]=='+'?1:0);
    }
    for(int i=1;i<=q;i++){
        scanf("%d%d",&nd[i].x,&nd[i].y);a[nd[i].x][nd[i].y]=0;
    }
    calc();
    nd[q].ans=maxx;
    for(int i=q;i>=2;i--){
        a[nd[i].x][nd[i].y]=1;
        calcc(nd[i].y);//更新 
        nd[i-1].ans=nd[i].ans;
        while(check(nd[i].x,maxx+1)) maxx++,nd[i-1].ans=maxx;//只改动了这一行 
    }
    for(int i=1;i<=q;i++) printf("%d\n",nd[i].ans); 
}
int main(){
    init();

    return 0;
}
