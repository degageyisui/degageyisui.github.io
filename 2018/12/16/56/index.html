#include<bits/stdc++.h>
#define maxn 205
#define maxm 2005
using namespace std;
int n,m,x,y,z;
struct eage{
    int from,to,next,cap,flow;
}e[maxm<<1];
int first[maxn],np=0;
void add(int u,int v,int c){
    e[++np]=(eage){u,v,first[u],c,0};
    first[u]=np;
    e[++np]=(eage){v,u,first[v],0,0};
    first[v]=np;
} 
int a[maxn],fa[maxn];
int other(int i){
    return i&1?i+1:i-1;
}
int bfs(int s,int t){
    queue<int>q;
    memset(a,0,sizeof(a));
    a[s]=0x3f3f3f3f;
    fa[s]=0;
    q.push(s);
    while(!q.empty()){
        int i=q.front();q.pop();
        for(int p=first[i];p;p=e[p].next){
            int j=e[p].to;
            if(!a[j]&&e[p].cap>e[p].flow){
                a[j]=min(a[i],e[p].cap-e[p].flow);
                fa[j]=p;
                q.push(j);
                if(j==t) return a[j];
            }
        }
    }
    return 0;//如果没有找到任意一条 
}
int ek(){
    int maxflow=0;
    while(1){
        int del=bfs(1,n);
        if(del==0) break;
        maxflow+=del;
        int f=fa[n];
        while(f){
            e[f].flow+=del;
            e[other(f)].flow-=del;
            f=fa[e[f].from];
        }
    }
    return maxflow;
}
void init(){
     scanf("%d%d",&m,&n);
     for(int i=1;i<=m;i++){
         scanf("%d%d%d",&x,&y,&z);
         add(x,y,z);
     }    
     printf("%d",ek());
}
int main(){
    init();

    return 0;
}

费用流:

#include<bits/stdc++.h>
#define maxn 305
#define maxm 3005
#define inf 0x3f3f3f3f
using namespace std;
struct eage{
    int from,to,next,cap,flow,w;
}e[maxm<<1];
int first[maxn],np=0;
void add(int u,int v,int c,int w){
    e[++np]=(eage){u,v,first[u],c,0,w};
    first[u]=np;
    e[++np]=(eage){v,u,first[v],0,0,-w};
    first[v]=np;
}
int other(int i){
    return i&1?i+1:i-1;
}
int n,m,x,y,z,w,dist[maxn],a[maxn],fa[maxn];
bool vis[maxn];
int spfa(){
    queue<int>q;
    memset(vis,0,sizeof(vis));
    memset(a,0,sizeof(a));
    a[1]=inf;
    fa[1]=0;q.push(1);
    memset(dist,0x3f,sizeof(dist));
    dist[1]=0;
    while(!q.empty()){
        int i=q.front();q.pop();
        vis[i]=0;
        for(int p=first[i];p;p=e[p].next){
            int j=e[p].to;
            if(e[p].cap>e[p].flow)
            if(dist[j]>dist[i]+e[p].w){
                dist[j]=dist[i]+e[p].w;
                a[j]=min(a[i],e[p].cap-e[p].flow);
                fa[j]=p;
                if(!vis[j]) q.push(j),vis[j]=1;
            }
        }
    }
    return a[n];
}
void ek(){
    int maxflow=0;
    int cost=0;
    while(spfa()){
        maxflow+=a[n];
        cost+=dist[n]*a[n];
        int f=fa[n];
        while(f){
            e[f].flow+=a[n];
            e[other(f)].flow-=a[n];
            f=fa[e[f].from];
        }
    }
    printf("%d %d",maxflow,cost);
}
void init(){
    scanf("%d%d",&m,&n);
    for(int i=1;i<=m;i++){
        scanf("%d%d%d%d",&x,&y,&z,&w);
        add(x,y,z,w);
    }
    ek();
}
int main(){
    init();

    return 0;
}
